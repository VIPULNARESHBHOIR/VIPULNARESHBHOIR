//GCD of numbers

#include <stdio.h>
int cal_GCD( int x1,int x2)
  { 
      if (x2!=0)
      return cal_GCD(x2,x1%x2);
      else
      return x1;
     
  }
  
int main()
{
  int x1,x2;
  printf("enter the two numbers to caclucate gcd::");
  scanf("%d%d",&x1,&x2);
  printf("\nGCD of %d & %d is %d\n\a",x1,x2,cal_GCD(x1,x2));
  return 0;
}
